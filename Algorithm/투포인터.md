=============
### 컴파일 언어 : PyPy3

count, sum=list(map(int, input().split(' ')))
a = list(map(int, input().split(' ')))

counting=0
temp=0

for i in range(count):
  temp=0
  for j in range(i, count):
    temp=temp + a[j]
    if(temp==sum):
      counting=counting+1
      break
    elif(temp>sum):
      break

print(counting)
